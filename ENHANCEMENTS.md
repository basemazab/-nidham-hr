\"employees:list\",\n  () => prisma.employee.findMany(),\n  300 // 5 دقائق TTL\n);\n\n// pagination\nconst page = 1;\nconst limit = 10;\nconst total = 100;\nconst pagination = calculatePaginationParams(page, limit, total);\n```\n\n### 3.2 تحسينات Prisma Queries\n\n**الملفات المتعلقة:**\n- `src/lib/api/performance-optimization.ts` — queryOptimizations\n\n**المميزات:**\n- ✅
